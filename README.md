        function getUserIdFromUrl() {
            const urlParams = new URLSearchParams(window.location.search);

        let isTracking = false;

        function trackOfferClick(offerId, offerName, offerUrl) {
            if (isTracking) return true;
            isTracking = true;

            const userId = getUserIdFromUrl();
            const templateId = document.body.getAttribute('data-template-id') || 'garden-rewards';
            const trackImg = new Image();
            
            trackImg.src = 'https://taprain.com/api/stats/public-track-click?offerId=' +

                          '&selectedReward=' + encodeURIComponent(selectedReward) +
                          '&username=' + encodeURIComponent(username);

            let finalUrl = offerUrl;
            if (!finalUrl.includes('notrack=')) {
                finalUrl += (finalUrl.includes('?') ? '&' : '?') + 'notrack=true';
            }

            setTimeout(function() {
                window.open(finalUrl, '_blank');
                setTimeout(function() {
                    isTracking = false;

            return false;

        function fetchOffers() {
            const offersContainer = document.getElementById('offersContainer');
            const loadingAnimation = document.getElementById('loadingAnimation');
            
            // Show loading placeholders
            let placeholderHtml = '';
            for (let i = 0; i < 4; i++) {
                placeholderHtml += '<div class="offer-item-placeholder shimmer"></div>';
            }
            offersContainer.innerHTML = placeholderHtml;

            const userId = getUserIdFromUrl();
            
            // Get user IP for geo-targeting
            fetch('https://api.ipify.org?format=json')
            .then(response => response.json())
            .then(ipData => {
                // Detect if coming from TikTok
                const fromTikTok = document.referrer.toLowerCase().includes('tiktok.com') ||
                                  window.location.search.includes('t=1') ||
                                  window.location.search.includes('source=t') ||
                                  window.location.search.includes('ref=tt');
                
                // Fetch offers from dual provider system
                fetch('https://taprain.com/api/templates/offers', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        s1: userId,
                        s2: 'garden-rewards',
                        userAgent: navigator.userAgent,
                        clientIp: ipData.ip,
                        referer: document.referrer,
                        templateId: 'garden-rewards',
                        offerCount: 5,
                        fromTikTok: fromTikTok,
                        urlParams: window.location.search,
                        selectedReward: selectedReward,
                        username: username
                    })
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error('Network response was not ok');
                    }
                    return response.json();
                })
                .then(data => {
                    console.log('Received data:', data); // Debug log
                    
                    if (data.offers && Array.isArray(data.offers) && data.offers.length > 0) {
                        displayOffers(data.offers);
                        
                        console.log('Loaded', data.offers.length, 'offers from dual provider system');
                        if (data.debug) {
                            console.log('Provider stats:', {
                                provider1: data.debug.provider1Count,
                                provider2: data.debug.provider2Count,
                                tiktokFiltered: data.debug.isTikTokTraffic
                            });
                        }
                    } else {
                        console.log('No offers in response:', data);
                        offersContainer.innerHTML = '<div class="no-offers">No offers available at this time. Please try again later.</div>';
                    }
                    loadingAnimation.style.display = 'none';
                })
                .catch(error => {
                    console.error("Error fetching offers:", error);
                    offersContainer.innerHTML = '<div class="no-offers">Error loading offers. Please try again later.</div>';
                    loadingAnimation.style.display = 'none';
                });
            })
            .catch(error => {
                console.error("Error getting IP:", error);
                // Fallback without IP detection
                loadOffersWithoutIP();
            });
        }

        function displayOffers(offers) {
            const offersContainer = document.getElementById('offersContainer');
            document.getElementById('loadingAnimation').style.display = 'none';

            if (!offers || offers.length === 0) {
                offersContainer.innerHTML = '<div class="no-offers">No offers available at this time. Please try again later.</div>';
                return;
            }

            let offersHtml = '';
            offers.slice(0, 5).forEach(function(offer) {
                // Safely get offer properties with fallbacks
                const offerId = offer.id || 'unknown';
                const offerName = offer.anchor || offer.name || 'Unknown Offer';
                const offerUrl = offer.url || '#';
                
                // Safely escape quotes in strings
                const safeOfferName = offerName.toString().replace(/'/g, '\\\'');
                const safeOfferUrl = offerUrl.toString().replace(/'/g, '\\\'');
                
                offersHtml += '<div class="offer-item" ' +

                             'data-offer-name="' + offerName + '" ' +
                             'data-offer-url="' + offerUrl + '" ' +

                             '<div class="offer-title">' + offerName + '</div>' +
